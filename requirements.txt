# -----------------------------------------------------------------------------
# [1] PyTorch — 必须手动选择与 CUDA 版本匹配的 wheel，不能依赖 pip 自动解析
# -----------------------------------------------------------------------------
#
# ★ 推荐：CUDA 12.8（PyTorch 2.7+ 官方支持，Blackwell/RTX 50xx 原生支持）
#   ⚠️  要求：NVIDIA 驱动 ≥ 560.x，GPU 架构 ≥ Turing（RTX 20xx / GTX 16xx，sm_75+）
#   ⚠️  不支持：Pascal 及更旧架构（GTX 10xx，sm_61 及以下）
#
#   pip install torch>=2.7.0 torchvision>=0.22.0 \
#       --index-url https://download.pytorch.org/whl/cu128
#
# ─── 向后兼容方案（RTX 30xx/40xx 以下或驱动版本较旧时） ──────────────────────
#
#   CUDA 12.1（兼容 RTX 20xx/30xx/40xx，驱动 ≥ 525.x）：
#   pip install torch>=2.0.0 torchvision>=0.15.0 \
#       --index-url https://download.pytorch.org/whl/cu121
#
#   CUDA 11.8（兼容 GTX 10xx 及更早 Pascal/Volta/Turing，驱动 ≥ 450.x）：
#   pip install torch>=2.0.0 torchvision>=0.15.0 \
#       --index-url https://download.pytorch.org/whl/cu118
#
# ─── CPU only（不支持 GPU 推理，仅用于调试） ─────────────────────────────────
#   pip install torch>=2.0.0 torchvision>=0.15.0
#
# ─── 选择依据速查 ─────────────────────────────────────────────────────────────
#   GPU 架构         sm 版本   推荐 CUDA wheel
#   RTX 50xx (B系)   sm_120    cu128（必须）
#   RTX 40xx         sm_89     cu128 或 cu121
#   RTX 30xx         sm_86     cu128 或 cu121
#   RTX 20xx/GTX 16xx sm_75   cu128 或 cu121
#   GTX 10xx         sm_61     cu118（cu128 不支持此架构）
#
# torch>=2.0.0 是 torch.compile 的最低版本要求（IFRNet / RealESRGAN 推理优化）
torch>=2.0.0
torchvision>=0.15.0

# 安装方式（三个包版本须与已安装 TensorRT 库严格匹配，建议统一通过 NVIDIA pip index）：
#
#   pip install tensorrt pycuda onnx onnxruntime-gpu \
#       --extra-index-url https://pypi.ngc.nvidia.com
#
# 说明：
#   tensorrt        — TRT Python 绑定（8.x 或 10.x，脚本已做双版本 API 兼容）
#   pycuda          — TRT Engine 执行所需的 CUDA 驱动绑定
#   onnx            — PyTorch 模型导出为 ONNX（TRT Engine 构建中间格式）
#   onnxruntime-gpu — ONNX 导出验证（可选，跳过则 TRT 构建时不做 ONNX 校验）
#
# ⚠️  注意：不要安装 pycuda.autoinit 或调用 pycuda.autoinit，
#     会与 PyTorch 的 CUDA context 冲突导致推理报错。
#

# pycuda
# onnx>=1.14.0
